-- (3) TABELA CLIENTE - OK! * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
--
CREATE TABLE Cliente (
codigoCli int  IDENTITY(1,1) PRIMARY KEY,
nome varchar(20) NOT NULL,
sobrenome varchar(20),
sexo char(1) CHECK (sexo IN ('M', 'F')),
cidade varchar(27) default 'Belo Horizonte',
estado char(2) default 'MG',
situacao char(1),
pontuacao int,
cpf bigint UNIQUE,
CONSTRAINT check_Pontuacao CHECK (pontuacao >= 0 and pontuacao <= 10),
)
GO
ALTER TABLE Cliente
ADD CONSTRAINT check_Situacao CHECK (situacao IN ('A', 'I'));
GO


-- (4) TABELA CONFERENTE - OK! * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
--
CREATE TABLE Conferente (
codigoConf int IDENTITY(1,1) PRIMARY KEY,
nome varchar(45) NOT NULL,
contato varchar(40)
)
GO

-- (5) TABELA UNIDADE - OK!  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * 
--
CREATE TABLE Unidade (
codigoUnid int IDENTITY(1,1) PRIMARY KEY,
sigla varchar(10) NOT NULL,
descricao varchar(35),
)
GO

-- (6) TABELA PRODUTO - OK! * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * 
--
CREATE TABLE Produto (
codigoPro int IDENTITY(1,1) PRIMARY KEY,
descricao varchar(60) NOT NULL,
quantidade int,
minimo int,
maximo int,
valorProduto decimal(10,2),
codigoUnid int,
FOREIGN KEY(codigoUnid) REFERENCES Unidade (codigoUnid)
)
GO
ALTER TABLE Produto ADD CONSTRAINT 
check_QuantidadeItem CHECK (quantidade >= 0);
ALTER TABLE Produto ADD CONSTRAINT 
check_minimo CHECK (minimo > 0);
ALTER TABLE Produto ADD CONSTRAINT 
check_maximo CHECK (maximo > 0);
ALTER TABLE Produto ADD CONSTRAINT 
check_ValorProduto CHECK (valorProduto > 0);
GO

-- (7) TABELA PEDIDO VALIDADA - OK! * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * 
--
CREATE TABLE Pedido (
numeroPed int IDENTITY(1,1) PRIMARY KEY,
codigoCli int,
codigoConf int,
data DateTime,
obs varchar(50),
valorTotal decimal(10,2),
FOREIGN KEY(codigoConf) REFERENCES Conferente (codigoConf),
FOREIGN KEY(codigoCli) REFERENCES Cliente (codigoCli),
)
GO

-- (8) TABELA ITEMPEDIDO - OK! * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
--
CREATE TABLE ItemPedido (
numeroItem int IDENTITY(1,1)  PRIMARY KEY,
numeroPed int,
codigoPro int,
valor decimal(10,2) CHECK (valor > 0),
quantidade int,
desconto decimal(10,2),
FOREIGN KEY(numeroPed) REFERENCES Pedido (numeroPed) ON DELETE CASCADE
)
ALTER TABLE ItemPedido ADD FOREIGN KEY(codigoPro) REFERENCES Produto (codigoPro)
GO

-- (9) TABELA MATERIALGERAL - OK * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
